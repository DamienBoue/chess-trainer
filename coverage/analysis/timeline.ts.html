function startOfWeek(ts: number): number {
  const d = new Date(ts)
  const day = (d.getUTCDay() + 6) % 7   // Monday = 0
  d.setUTCHours(0, 0, 0, 0)
  d.setUTCDate(d.getUTCDate() - day)
  return d.getTime()
}
 
function shortLabel(ms: number): string {
  const d = new Date(ms)
  return `${d.getUTCDate()}/${d.getUTCMonth() + 1}`
}
